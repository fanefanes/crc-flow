# Installation

## npm

``` sh
npm install crc-flow --save
```

# Usage

## Flow

``` 

import Editor, { Flow } from 'crc-flow';

const data = {
  nodes: [
    {
      id: '0',
      comboId: 'combo1',
      label: 'Node',
      x: 55,
      y: 55,
    },
    {
      id: '1',
      comboId: 'combo1',
      label: 'Node',
      x: 55,
      y: 255,
    },
  ],
  edges: [
    {
      label: 'Label',
      source: '0',
      target: '1',
    },
  ],
  combo: [
    {
      id: 'combo1'
    }
  ]
};

<Editor>
  <Flow data={data} />
</Editor>

```
